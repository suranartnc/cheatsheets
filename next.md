## Routing

{: .-two-column}

### Routes

```bash
File: ./routes.js
```
{: .-setup}

```js
const routes = [
  {
    name: 'home',
    pattern: '/',
    page: 'index'
  }, {
    name: 'about',
    pattern: '/about/',
  }, {
    name: 'entry',
    pattern: '/:id(\\d+)/',
  }
]
```


### Link

```js
import { Link } from './routes'
```
{: .-setup}

```js
function MyComponent() {
  return (
    <div>
      <Link route="home">
        <a>Home</a>
      </Link>
      <Link route="entry" params={{ id: 1234 }}>
        <a>Entry: 1234</a>
      </Link>
    </div>
  )
}
```
